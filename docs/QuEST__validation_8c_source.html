98 int isMatrixUnitary(ComplexMatrix2 u) {
99  if ( absReal( u.r0c0.real*u.r0c0.real
100  + u.r0c0.imag*u.r0c0.imag
101  + u.r1c0.real*u.r1c0.real
102  + u.r1c0.imag*u.r1c0.imag - 1) > REAL_EPS ) return 0;
103  if ( absReal( u.r0c1.real*u.r0c1.real
104  + u.r0c1.imag*u.r0c1.imag
105  + u.r1c1.real*u.r1c1.real
106  + u.r1c1.imag*u.r1c1.imag - 1) > REAL_EPS ) return 0;
107  if ( absReal( u.r0c0.real*u.r0c1.real
108  + u.r0c0.imag*u.r0c1.imag
109  + u.r1c0.real*u.r1c1.real
110  + u.r1c0.imag*u.r1c1.imag) > REAL_EPS ) return 0;
111  if ( absReal( u.r0c1.real*u.r0c0.imag
112  - u.r0c0.real*u.r0c1.imag
113  + u.r1c1.real*u.r1c0.imag
114  - u.r1c0.real*u.r1c1.imag) > REAL_EPS ) return 0;
115  return 1;
116 }
