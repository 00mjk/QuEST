62 REAL calcTotalProbability(MultiQubit multiQubit){
63  // implemented using Kahan summation for greater accuracy at a slight floating
64  // point operation overhead. For more details see https://en.wikipedia.org/wiki/Kahan_summation_algorithm
65  REAL pTotal=0;
66  REAL y, t, c;
67  long long int index;
68  long long int numAmpsPerRank = multiQubit.numAmpsDividedByNumChunks;
69  c = 0.0;
70  for (index=0; index<numAmpsPerRank; index++){
71  // Perform pTotal+=multiQubit.stateVec.real[index]*multiQubit.stateVec.real[index]; by Kahan
72 
73  y = multiQubit.stateVec.real[index]*multiQubit.stateVec.real[index] - c;
74  t = pTotal + y;
75  // Don't change the bracketing on the following line
76  c = ( t - pTotal ) - y;
77  pTotal = t;
78 
79  // Perform pTotal+=multiQubit.stateVec.imag[index]*multiQubit.stateVec.imag[index]; by Kahan
80 
81  y = multiQubit.stateVec.imag[index]*multiQubit.stateVec.imag[index] - c;
82  t = pTotal + y;
83  // Don't change the bracketing on the following line
84  c = ( t - pTotal ) - y;
85  pTotal = t;
86 
87 
88  }
89  return pTotal;
90 }

211 int measureWithStats(MultiQubit multiQubit, int measureQubit, REAL *stateProb){
212  QuESTAssert(measureQubit >= 0 && measureQubit < multiQubit.numQubits, 2, __func__);
213 
214  int outcome;
215  // find probability of qubit being in state 1
216  REAL stateProbInternal = findProbabilityOfOutcome(multiQubit, measureQubit, 1);
217 
218  // we can't collapse to a state that has a probability too close to zero
219  if (stateProbInternal<REAL_EPS) outcome=0;
220  else if (1-stateProbInternal<REAL_EPS) outcome=1;
221  else {
222  // ok. both P(0) and P(1) are large enough to resolve
223  // generate random float on [0,1]
224  float randNum = genrand_real1();
225  if (randNum<=stateProbInternal) outcome = 1;
226  else outcome = 0;
227  }
228  if (outcome==0) stateProbInternal = 1-stateProbInternal;
229  collapseToOutcomeLocal(multiQubit, measureQubit, stateProbInternal, outcome);
230  *stateProb = stateProbInternal;
231  return outcome;
232 }
