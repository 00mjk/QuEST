62 int generateMeasurementOutcome(REAL zeroProb, REAL *outcomeProb) {
63 
64  // randomly choose outcome
65  int outcome;
66  if (zeroProb < REAL_EPS)
67  outcome = 1;
68  else if (1-zeroProb < REAL_EPS)
69  outcome = 0;
70  else
71  outcome = (genrand_real1() > zeroProb);
72 
73  // set probability of outcome
74  if (outcome == 0)
75  *outcomeProb = zeroProb;
76  else
77  *outcomeProb = 1 - zeroProb;
78 
79  return outcome;
80 }

92 void seedQuESTDefault(){
93  // init MT random number generator with three keys -- time, pid and a hash of hostname
94  // for the MPI version, it is ok that all procs will get the same seed as random numbers will only be
95  // used by the master process
96 
97  struct timeval tv;
98  gettimeofday(&tv, NULL);
99 
100  double time_in_mill =
101  (tv.tv_sec) * 1000 + (tv.tv_usec) / 1000 ; // convert tv_sec & tv_usec to millisecond
102 
103  unsigned long int pid = getpid();
104  unsigned long int msecs = (unsigned long int) time_in_mill;
105  char hostName[MAXHOSTNAMELEN+1];

107  unsigned long int hostNameInt = hashString(hostName);
108 
109  unsigned long int key[3];

112 }
