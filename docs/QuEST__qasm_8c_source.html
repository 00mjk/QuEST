51 void qasm_setup(QubitRegister* qureg) {
52 
53  // populate and attach QASM logger
54  QASMLogger *qasmLog = malloc(sizeof *qasmLog);
55  qureg->qasmLog = qasmLog;
56  if (qasmLog == NULL)
57  bufferOverflow();
58 
59  qasmLog->isLogging = 0;
60  qasmLog->bufferSize = BUF_INIT_SIZE;
61  qasmLog->buffer = malloc(qasmLog->bufferSize * sizeof *(qasmLog->buffer));
62  if (qasmLog->buffer == NULL)
63  bufferOverflow();
64 
65  // add headers and quantum / classical register creation
66  qasmLog->bufferFill = snprintf(
67  qasmLog->buffer, qasmLog->bufferSize,
68  "OPENQASM 2.0;\nqreg %s[%d];\ncreg %s[%d];\n",
69  QUREG_LABEL, qureg->numQubitsRepresented,
70  MESREG_LABEL, qureg->numQubitsRepresented);
71  if (qasmLog->bufferFill >= qasmLog->bufferSize)
72  bufferOverflow();
73 }

83 void addStringToQASM(QubitRegister qureg, char line[], int lineLen) {
84 
85  char* buf = qureg.qasmLog->buffer;
86  int bufSize = qureg.qasmLog->bufferSize;
87  int bufFill = qureg.qasmLog->bufferFill;
88 
89  // grow QASM buffer if necessary
90  if (lineLen + bufFill > bufSize) {
91 
92  int newBufSize = BUF_GROW_FAC * bufSize;
93  if (lineLen + bufFill > newBufSize)
94  bufferOverflow();
95 
96  char* newBuffer = malloc(newBufSize * sizeof *newBuffer);
97  sprintf(newBuffer, "%s", buf);
98  free(buf);
99 
100  qureg.qasmLog->bufferSize = newBufSize;
101  qureg.qasmLog->buffer = newBuffer;
102  bufSize = newBufSize;
103  buf = newBuffer;
104  }
105 
106  // add new str
107  int addedChars = snprintf(buf+bufFill, bufSize-bufFill, "%s", line);
108  qureg.qasmLog->bufferFill += addedChars;
109 }

111 void addGateToQASM(QubitRegister qureg, TargetGate gate, int* controlQubits, int numControlQubits, int targetQubit, REAL* params, int numParams) {
112 
113  int len = 0;
114  char line[MAX_LINE_LEN + 1]; // for trailing \0
115 
116  // add control labels
117  for (int i=0; i < numControlQubits; i++)
118  len += snprintf(line+len, MAX_LINE_LEN-len, "%s", CTRL_LABEL_PREF);
119 
120  // add target gate
121  len += snprintf(line+len, MAX_LINE_LEN-len, "%s", qasmGateLabels[gate]);
122 
123  // add argument if exists
124  if (numParams > 0) {
125  len += snprintf(line+len, MAX_LINE_LEN-len, "(");
126  for (int i=0; i < numParams; i++) {
127  len += snprintf(line+len, MAX_LINE_LEN-len, REAL_STRING_FORMAT, params[i]);
128  if (i != numParams - 1)
129  len += snprintf(line+len, MAX_LINE_LEN-len, ",");
130  }
131  len += snprintf(line+len, MAX_LINE_LEN-len, ")");
132  }
133 
134  // add space
135  len += snprintf(line+len, MAX_LINE_LEN-len, " ");
136 
137  // add control qubits
138  for (int i=0; i < numControlQubits; i++)
139  len += snprintf(line+len, MAX_LINE_LEN-len, "%s[%d],", QUREG_LABEL, controlQubits[i]);
140 
141  // add target qubit, colon and newline
142  len += snprintf(line+len, MAX_LINE_LEN-len, "%s[%d];\n", QUREG_LABEL, targetQubit);
143 
144  // check whether we overflowed buffer
145  if (len >= MAX_LINE_LEN)
146  bufferOverflow();
147 
148  addStringToQASM(qureg, line, len);
149 }
